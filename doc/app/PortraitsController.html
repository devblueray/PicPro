def create
  @portrait = Portrait.new(portrait_params)
  @gallery = Gallery.where(name: params[:gallery_id])

  if @portrait.save
    redirect_to :back, notice: "The portrait #{@portrait.name} has been uploaded."
  else
    render "new"
  end
end
